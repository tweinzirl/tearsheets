CREATE TABLE accounts (
    ID integer,
    Client_ID integer,
    Product_Group varchar(128),
    Acct_Nbr varchar(128),
    Product_Type varchar(128),
    Balance real,
    Banker_ID integer,
    Primary_Banker integer,
    Start_Dt varchar(128),
    End_Dt varchar(128),
    Is_Current integer,
    Status varchar(128),
    PRIMARY KEY (ID),
    FOREIGN KEY (Client_ID) REFERENCES clients(ID),
    FOREIGN KEY (Banker_ID) REFERENCES bankers(ID)
);

CREATE TABLE clients (
    ID integer,
    Person_Org varchar(128),
    Type varchar(128),
    Product_Mix varchar(128),
    Name varchar(128),
    First_Name varchar(128),
    Last_Name varchar(128),
    NAICS_CD varchar(128),
    NAICS_CD_DESC varchar(128),
    SSN varchar(128),
    Employer varchar(128),
    Title varchar(128),
    Address varchar(128),
    Region varchar(128),
    Birthday varchar(128),
    Start_Dt varchar(128),
    End_Dt varchar(128),
    Is_Current integer,
    Wealth_Tier varchar(128),
    HH_ID integer,
    PRIMARY KEY (ID),
    FOREIGN KEY (HH_ID) REFERENCES links(Household_ID)
);

CREATE TABLE bankers (
    ID integer,
    Login varchar(128),
    Name varchar(128),
    Type varchar(128),
    Branch_Nbr integer,
    Region varchar(128),
    Start_Dt varchar(128),
    End_Dt varchar(128),
    Is_Current integer,
    PRIMARY KEY (ID),
    FOREIGN KEY (Branch_Nbr) REFERENCES branches(Branch_Number)
);

CREATE TABLE links (
    ID integer,
    Household_ID integer,
    Client_1 integer,
    Client_2 integer,
    Link_Type varchar(128),
    PRIMARY KEY (ID),
    FOREIGN KEY (Household_ID) REFERENCES clients(HH_ID)
);

CREATE TABLE transactions (
    ID integer,
    Acct_Nbr varchar(128),
    Prod_Dt integer,
    Balance real,
    Transaction_Amount real,
    After_Tran_Balance real,
    Transaction_Type varchar(128),
    Transaction_Purpose varchar(128),
    Transaction_Description varchar(128),
    PRIMARY KEY (ID),
    FOREIGN KEY (Acct_Nbr) REFERENCES accounts(Acct_Nbr)
);

CREATE TABLE recommendations (
    ID integer,
    Client_ID integer,
    Report_Name varchar(128),
    Report_ID integer,
    Is_Top_3 integer,
    Recommendation_Text varchar(128),
    PRIMARY KEY (ID),
    FOREIGN KEY (Client_ID) REFERENCES clients(Client_ID)
);
